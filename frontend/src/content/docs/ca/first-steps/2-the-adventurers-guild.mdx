import QuestionCard from "@components/QuestionCard.astro";
import CodeBlock from "@components/CodeBlock/CodeBlock.svelte";

export function SimpleQuestion({
  question,
  code,
  answer,
  correct,
  wrong,
  checkMessed = true,
}) {
  const no_dummy = code.toString().replace("?", "");
  const setup = `
        const VALUE: &str = "__VALUE__";
        if ${checkMessed} && !VALUE.starts_with("${no_dummy}") {
            println!("Sembla que has modificat el codi, fes clic al botó de \\"Reset\\" per tornar-lo al seu estat original.");
            println!("Recorda modificar només les parts amb ?.");
            return;
        }
        let a = VALUE.replace("${no_dummy}", "");
        match a.replace(" ", "").as_str() {
            "${answer}" => println!("${correct}"),
            "?" => println!("Substitueix ? amb la teva resposta."),
            _ => println!("${wrong}"),
        }
    `.replaceAll("\n", "");

return (

<QuestionCard question={question}>
<CodeBlock
        client:visible
        code={code}
        setup={setup}
        showLineNumbers={false}
        lang="ca"
      />
</QuestionCard>
);
}
export function FreeQuestion({ question, code = "", correct }) {
const setup = `         if __VALUE__ {
            println!("${correct}");
        } else {
            println!("La teva resposta és incorrecta, torna a intentar-ho!");
        }
    `.replaceAll("\n", "");
return (
<QuestionCard question={question}>
<CodeBlock
        client:visible
        code={code}
        setup={setup}
        showLineNumbers={false}
        lang="ca"
      />
</QuestionCard>
);
}

Bon dia, benvingut a Newlia, on neixen els herois!

Estas aqui per registrar-te com a aventurer? I vols escollir la classe de "Programador"?
Bona elecció! Els programadors son una classe versàtil, dóna'ls suficient temps i poden fer el que sigui!

Aquesta classe requereix una petita prova, per assegurar que tens les habilitats mínimes per resoldre problemes.  
Només et portarà un moment.

Si us plau, emplena els espais dels següents enunciats, substituïnt **`?`** amb la resposta correcta.

:::note[No et preocupis]
Si t'equivoques amb el codi, fes clic al botó de "Reset" per tornar-lo al seu estat original.
:::
