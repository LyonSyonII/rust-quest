import QuestionCard from "@components/QuestionCard.astro";
import CodeBlock from "@components/CodeBlock/CodeBlock.svelte";

export function SimpleQuestion({
  question,
  code,
  answer,
  correct,
  wrong,
  checkMessed = true,
}) {
  const no_dummy = code.toString().replace("?", "");
  const setup = `
        const VALUE: &str = "__VALUE__";
        if ${checkMessed} && !VALUE.starts_with("${no_dummy}") {
            println!("Parece que has estropeado el código, haz clic en el botón \\"Reset\\" para devolverlo a su estado original.");
            println!("Recuerda modificar únicamente las partes con ?.");
            return;
        }
        let a = VALUE.replace("${no_dummy}", "");
        match a.replace(" ", "").as_str() {
            "${answer}" => println!("${correct}"),
            "?" => println!("Substituye ? con tu respuesta."),
            _ => println!("${wrong}"),
        }
    `.replaceAll("\n", "");
  return (
    <QuestionCard question={question}>
      <CodeBlock
        client:visible
        code={code}
        setup={setup}
        showLineNumbers={false}
      />
    </QuestionCard>
  );
}
export function FreeQuestion({ question, code = "", correct }) {
  const setup = `
        if __VALUE__ {
            println!("${correct}");
        } else {
            println!("Tu respuesta es incorrecta, ¡inténtalo de nuevo!");
        }
    `.replaceAll("\n", "");
  return (
    <QuestionCard question={question}>
      <CodeBlock
        client:visible
        code={code}
        setup={setup}
        showLineNumbers={false}
        lang="es"
      />
    </QuestionCard>
  );
}
