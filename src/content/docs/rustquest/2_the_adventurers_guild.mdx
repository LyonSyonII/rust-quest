import CodeBlock from "../../../components/CodeBlock.svelte";
export function SimpleQuestion({ code, answer, correct, wrong, checkMessed = true }) {
  console.log(code, answer, correct, wrong);
  const no_dummy = code.toString().replace("?", "");
  const setup = `
        const VALUE: &str = "__VALUE__";
        if ${checkMessed} && !VALUE.starts_with("${no_dummy}") {
            println!("Seems like you've messed up the code, click the \\"Reset\\" button to return it back to its original state.");
            println!("Remember to only modify the ? part of the code.");
            return;
        }
        let a = VALUE.replace("${no_dummy}", "");
        match a.trim() {
            "${answer}" => println!("${correct}"),
            "?" => println!("Replace ? with your answer."),
            _ => println!("${wrong}"),
        }
    `.replaceAll("\n", "");
  return <CodeBlock client:only code={code} setup={setup} />;
}
export function FreeQuestion({ code, correct }) {
    const setup = `
        const VALUE: &str = "__VALUE__";
        if VALUE.contains("?") {
            println!("Replace ? with your answer.");
            return;
        }
        if __VALUE__ {
            println!("${correct}");
        } else {
            println!("Your answer is wrong, try again!");
        }
    `.replaceAll("\n", "");
    const errorMsg = "Woops, something went wrong and the code does not compile!\n\nIf you've mistakenly messed up the code, click the \"Reset\" button to return it back to its original state.\n\nRemember to replace ? with your answer.";
    return <CodeBlock client:only code={code} setup={setup} errorMsg={errorMsg} />;
}
